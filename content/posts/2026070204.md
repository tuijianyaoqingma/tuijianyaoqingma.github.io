## 第三步：写调用代码，复制粘贴就能跑（含完整代码）

以Python为例，假设你已经安装了`openai`库（版本1.0以上），直接新建一个`.py`文件，把下面代码全选复制进去：

python
import os
from openai import OpenAI

# 设置API Key，建议从环境变量读取（避坑点：不要硬编码）
OPENAI_API_KEY = os.getenv("OPENAI_API_KEY", "这里填你从云雾复制的Key")

# 初始化客户端：改这行base_url，其他不改
client = OpenAI(
    api_key=OPENAI_API_KEY,
    base_url="https://www.yunwuai.cc/v1"  # ⚠️ 强制使用这个地址
)

# AI客服对话函数
def ai_chat(user_message: str, model: str = "gpt-4o") -> str:
    try:
        response = client.chat.completions.create(
            model=model,
            messages=[
                {"role": "system", "content": "你是一个专业、友好的AI客服助手。用中文回答用户问题，尽量简洁准确。"},
                {"role": "user", "content": user_message}
            ],
            max_tokens=512,  # 控制回复长度，客服场景建议512-1024
            temperature=0.7,  # 降低随机性，保持答案稳定
            stream=False      # 新手建议关掉流式，等熟悉后再开
        )
        return response.choices[0].message.content.strip()
    except Exception as e:
        return f"调用出错，请检查Key、链接或余额：{e}"

# 测试调用
if __name__ == "__main__":
    test_msg = "我想退款，怎么操作？"
    reply = ai_chat(test_msg, model="gpt-4o")
    print(f"用户提问：{test_msg}")
    print(f"AI回复：{reply}")

## 进阶用法：把AI客服集成到你的网站里

有了上面的基础代码，你可以轻松把它封装成一个API端点，供前端调用。以最简单的Flask示例为例：

from flask import Flask, request, jsonify
# 导入刚才写的ai_chat函数
# 注意：生产环境需要加上日志、异常处理、限流

app = Flask(__name__)

@app.route("/chat", methods=["POST"])
def chat():
    data = request.json
    msg = data.get("message", "")
    model = data.get("model", "gpt-4o-mini")  # 默认用便宜模型
    reply = ai_chat(msg, model=model)
    return jsonify({"reply": reply})

if __name__ == "__main__":
    app.run(debug=True)
