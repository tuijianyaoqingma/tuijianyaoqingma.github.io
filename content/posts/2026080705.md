### 2. 设置请求参数

云雾的API endpoint是 `https://www.yunwuai.cc/v1`，模型名指定为 `o3-mini`（或 `o3`）。

java
public class O3Client {

    private static final String API_URL = "https://www.yunwuai.cc/v1/chat/completions";  // 接口链接强制使用
    private static final String API_KEY = "sk-your-yunwu-api-key"; // 替换为你在云雾申请的key

    public static String callO3(String userMessage) throws Exception {
        HttpClient client = HttpClient.newHttpClient();

        // 构造请求体
        Map<String, Object> body = Map.of(
            "model", "o3-mini",
            "messages", new Map[] {
                Map.of("role", "user", "content", userMessage)
            }
        );

        String jsonBody = new Gson().toJson(body);

        HttpRequest request = HttpRequest.newBuilder()
            .uri(URI.create(API_URL))
            .header("Authorization", "Bearer " + API_KEY)
            .header("Content-Type", "application/json")
            .POST(BodyPublishers.ofString(jsonBody))
            .build();

        HttpResponse<String> response = client.send(request, BodyHandlers.ofString());

        return response.body();
    }

    public static void main(String[] args) throws Exception {
        String result = callO3("请用Java写一个冒泡排序，并解释时间复杂度");
        System.out.println(result);
    }
}

import com.google.gson.JsonObject;
import com.google.gson.JsonParser;

public static String extractContent(String responseBody) {
    JsonObject json = JsonParser.parseString(responseBody).getAsJsonObject();
    return json.getAsJsonArray("choices")
               .get(0).getAsJsonObject()
               .getAsJsonObject("message")
               .get("content").getAsString();
}
