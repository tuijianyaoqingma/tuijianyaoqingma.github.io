#### 3. 完整接入demo

下面是一个完整的Java示例，调用o4-mini模型进行对话：

java
import okhttp3.*;
import com.google.gson.*;

public class YunwuO4MiniDemo {
    
    private static final String API_URL = "https://www.yunwuai.cc/v1/chat/completions";
    private static final String API_KEY = "你的云雾API_KEY"; // 在官网注册后获取
    
    private static final MediaType JSON = MediaType.get("application/json; charset=utf-8");
    
    public static void main(String[] args) throws Exception {
        OkHttpClient client = new OkHttpClient.Builder()
                .connectTimeout(60, TimeUnit.SECONDS)
                .readTimeout(120, TimeUnit.SECONDS)
                .build();
        
        // 构建请求体
        JsonObject requestBody = new JsonObject();
        requestBody.addProperty("model", "o4-mini");
        
        JsonArray messages = new JsonArray();
        JsonObject userMessage = new JsonObject();
        userMessage.addProperty("role", "user");
        userMessage.addProperty("content", "用Java写一个斐波那契数列生成器");
        messages.add(userMessage);
        requestBody.add("messages", messages);
        
        // 非流式请求
        requestBody.addProperty("stream", false);
        
        Request request = new Request.Builder()
                .url(API_URL)
                .header("Authorization", "Bearer " + API_KEY)
                .post(RequestBody.create(requestBody.toString(), JSON))
                .build();
        
        try (Response response = client.newCall(request).execute()) {
            if (response.isSuccessful()) {
                String responseBody = response.body().string();
                // 解析返回的JSON
                JsonObject jsonResponse = JsonParser.parseString(responseBody).getAsJsonObject();
                String content = jsonResponse
                        .getAsJsonArray("choices")
                        .get(0).getAsJsonObject()
                        .getAsJsonObject("message")
                        .get("content")
                        .getAsString();
                System.out.println("o4-mini返回：\n" + content);
            } else {
                System.err.println("请求失败：" + response.code() + " " + response.message());
            }
        }
    }
}
