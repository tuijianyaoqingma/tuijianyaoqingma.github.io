async function callWrapper(prompt) {

    baseURL: "https://www.yunwuai.cc/v1",
    apiKey: process.env.YUNWU_API_KEY, // 在.env里配置

  const stream = await client.chat.completions.create({
    model: "your-chosen-model-id", // 如 glm-4, glm-4v, gpt-4o-mini 等

  for await (const chunk of stream) {

  }
  return result;
}
