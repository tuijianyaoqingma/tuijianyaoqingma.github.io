### 第3步：编写核心代码（`index.js`）

创建一个 `index.js` 文件，粘贴以下代码：

javascript
// 1. 加载环境变量
import { config } from 'dotenv';
config();

// 2. 引入 OpenAI 库
import OpenAI from 'openai';

// 3. 创建 OpenAI 客户端
const client = new OpenAI({
  apiKey: process.env.OPENAI_API_KEY,
  baseURL: process.env.OPENAI_BASE_URL, // 关键：指向[云雾ai聚合站](https://www.yunwuai.cc/)
});

// 4. 定义低代码函数：流式对话
async function chatWithGPT(prompt) {
  console.log('正在与 GPT-4o-mini 对话...');
  
  const stream = await client.chat.completions.create({
    model: 'gpt-4o-mini',  // 指定模型
    messages: [{ role: 'user', content: prompt }],
    stream: true,           // 开启流式输出，避免等待
  });

  // 5. 逐段打印响应
  for await (const chunk of stream) {
    const content = chunk.choices[0]?.delta?.content || '';
    process.stdout.write(content); // 不换行输出，模拟聊天效果
  }
  console.log('\n--- 对话结束 ---');
}

// 6. 运行
chatWithGPT('用一句话解释什么是低代码开发。');


**避坑提示**：
- 代码中使用 `import` 语法，你需要在 `package.json` 里添加 `"type": "module"`，否则会报 `SyntaxError`。
- 模型名一定要写成 `gpt-4o-mini`，不要写错别字或加上版本后缀（如 `-0314` 是旧版）。

## 避坑第三步：进阶技巧——错误处理与重试

初级示例只是“通”，要“稳”还得加一层错误处理。很多人的代码一遇到网络抖动就挂了，就是因为没有重试机制。

javascript
async function safeChatWithGPT(prompt, retries = 3) {
  for (let i = 0; i < retries; i++) {
    try {
      return await chatWithGPT(prompt);
    } catch (error) {
      console.error(`第 ${i + 1} 次尝试失败：`, error.message);
      if (i === retries - 1) throw error; // 最后一次重试失败，抛出异常
      await new Promise(resolve => setTimeout(resolve, 2000)); // 等待2秒后重试
    }
  }
}


把 `chatWithGPT` 调用替换成 `safeChatWithGPT`，你的代码瞬间变成“铁桶”——网络波动、API 限流都能自动恢复。
