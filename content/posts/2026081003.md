即使有了云雾这个稳定通道，网络也难免偶发波动。建议加个简单的重试逻辑，但不用像直连时那样频繁重试：

javascript
async function callWithRetry(prompt, retries = 3) {
  for (let i = 0; i < retries; i++) {
    try {
      return await callO3Mini(prompt);
    } catch (err) {
      console.error(`第${i+1}次失败：${err.message}`);
      if (i < retries - 1) await new Promise(r => setTimeout(r, 1000));
    }
  }
  throw new Error('重试后仍然失败');
}
