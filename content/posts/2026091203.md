import com.theokanning.openai.completion.chat.ChatCompletionRequest;
import com.theokanning.openai.completion.chat.ChatMessage;
import java.util.List;

public class GLM4Example {

    // 在此设置你的API Key，从[云雾ai聚合站](https://www.yunwuai.cc/)控制台获取

    public static void main(String[] args) {

        OpenAiService service = new OpenAiService(API_KEY, BASE_URL, Duration.ofSeconds(30));

        ChatCompletionRequest request = ChatCompletionRequest.builder()
                .model("gpt-4o-mini")   // 我们示例直接用“默认分组”下的模型；实际可换为智谱清言GLM-4

                        new ChatMessage("system", "你是一个资深Java工程师。请给出代码优化建议。"),
                        new ChatMessage("user", "如何用Java实现一个高性能的文本摘要API？请提供核心代码片段。")
                ))
                .maxTokens(1024)
                .temperature(0.7)
                .build();

        try {
            String response = service.createChatCompletion(request)
                    .getChoices().get(0).getMessage().getContent();
            System.out.println("智谱清言（通过云雾API）返回结果：\n" + response);
        } catch (Exception e) {
            System.out.println("调用失败：" + e.getMessage());
        }
    }
}
