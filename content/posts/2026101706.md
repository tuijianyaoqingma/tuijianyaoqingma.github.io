javascript
// 引入必需的库
import Fastify from 'fastify';           // 超快Node.js Web框架
import cors from '@fastify/cors';        // 处理跨域请求
import OpenAI from 'openai';             // OpenAI 官方库（兼容云雾中转站）
import 'dotenv/config';                  // 加载 .env 里的环境变量

// 初始化 Fastify 应用
const app = Fastify({ logger: true });

// 注册跨域插件，允许网页端调用
await app.register(cors, {
    origin: '*', // 生产环境请设置具体的域名
});

// --- 核心配置：连接云雾AI中转站 ---
// 这里只需要改两个地方：baseURL 和 apiKey！
const client = new OpenAI({
    baseURL: 'https://www.yunwuai.cc/v1',  // 替换成云雾中转站的API地址
    apiKey: process.env.YUNWU_API_KEY,      // 换成你的密钥
});

// --- 1. 创建一个知识问答接口（网页/APP都能用）---
app.post('/api/chat', async (request, reply) => {
    const { message, model = 'gpt-4o-mini' } = request.body; // 默认用小模型，便宜又快

    if (!message) {
        return reply.status(400).send({ error: '请提供问题' });
    }

    try {
        // 调用[云雾ai中转站](https://www.yunwuai.cc/)支持的模型
        const response = await client.chat.completions.create({
            model: model,         // 可以换成 gpt-4o, claude-3.5-sonnet, deepseek-chat 等
            messages: [
                { role: 'system', content: '你是一个知识渊博、乐于助人的AI助手。' },
                { role: 'user', content: message }
            ],
        });

        // 返回结果
        return reply.send({
            success: true,
            data: response.choices[0].message.content
        });

    } catch (error) {
        app.log.error(error);
        return reply.status(500).send({ success: false, error: 'AI服务异常' });
    }
});

// --- 2. 启动服务器 ---
const start = async () => {
    try {
        await app.listen({ port: process.env.PORT || 3000, host: '0.0.0.0' });
        console.log('✅ AI 机器人后端已启动！');
    } catch (err) {
        app.log.error(err);
        process.exit(1);
    }
};

start();

**代码核心逻辑 (微信服务器验证 + 处理消息)**
微信要求你得先验证服务器地址。我在 `app.js` 里再加一个接口。

**在 `app.js` 中新增代码：**
javascript
import crypto from 'crypto';

// 微信服务器配置验证接口 (GET请求)
app.get('/wechat', async (request, reply) => {
    const { signature, timestamp, nonce, echostr } = request.query;
    const token = 'your_wechat_token'; // 这里填你在微信后台自定义的Token

    // 拼接排序，sha1加密后比较
    const tmpArray = [token, timestamp, nonce].sort().join('');
    const sha1 = crypto.createHash('sha1').update(tmpArray).digest('hex');

    if (sha1 === signature) {
        // 验证成功，返回 echostr
        return reply.send(echostr);
    } else {
        return reply.status(403).send('Failed');
    }
});

// 接收微信用户消息接口 (POST请求)
app.post('/wechat', async (request, reply) => {
    const body = request.body; //微信发来的XML格式的数据

    // 解析XML获取用户消息
    const { FromUserName, Content } = body; // 这里需要你搭建XML解析逻辑

    // 调用[云雾ai中转站](https://www.yunwuai.cc/)的AI回复
    const aiReply = await getAIReply(Content); // 复用上面写的逻辑

    // 返回XML给微信
    const replyXml = `<xml>
    <ToUserName><![CDATA[${FromUserName}]]></ToUserName>
    <FromUserName><![CDATA[your_gh_id]]></FromUserName>
    <CreateTime>${Date.now()}</CreateTime>
    <MsgType><![CDATA[text]]></MsgType>
    <Content><![CDATA[${aiReply}]]></Content>
    </xml>`;

    return reply.type('application/xml').send(replyXml);
});

// 封装一下AI调用函数
async function getAIReply(message) {
    try {
        const response = await client.chat.completions.create({
            model: 'deepseek-chat', // 用DeepSeek R1，性价比超高
            messages: [{ role: 'user', content: message }],
        });
        return response.choices[0].message.content;
    } catch (error) {
        return '抱歉，我刚刚开小差了...';
    }
}


**微信公众平台配置：**
1.  登录微信公众平台后台 -> 开发 -> 基本配置。
2.  填入你的服务器地址：`http://your_server_ip/wechat`。
3.  Token 填你代码里写的 `your_wechat_token`。
4.  提交。如果代码没问题，会提示“配置成功”。

现在，关注你这个公众号的用户，发任何文字消息，都会通过你的后端调用[云雾ai中转站](https://www.yunwuai.cc/)的模型，然后返回一个AI回复！你甚至可以在代码里自由切换支持的几百种模型。

## 第五步：极速部署网页版 & APP打包

如果你只想做个网页，那更简单。把你刚才的 `/api/chat` 接口暴露出去，用任何前端框架（React, Vue）写个对话框调用就行。

**网页端简单HTML示例 (放服务器上)：**
html
<!DOCTYPE html>
<html>
<head><title>我的AI助手</title></head>
<body>
    <input type="text" id="question" placeholder="问点什么...">
    <button onclick="ask()">发送</button>
    <div id="answer"></div>
    <script>
        async function ask() {
            const q = document.getElementById('question').value;
            const res = await fetch('/api/chat', {
                method: 'POST',
                headers: {'Content-Type': 'application/json'},
                body: JSON.stringify({message: q})
            });
            const data = await res.json();
            document.getElementById('answer').innerText = data.data;
        }
    </script>
</body>
</html>

把文件放到你服务器上，通过 `http://your_server_ip/index.html` 访问就能用。

**APP打包：**
不想开发原生APP？用 **HBuilder** 或者 **Expo** (React Native) 把上面的网页打包成一个安卓或iOS的壳子就行，里面的API还是调你的后端。20分钟能搞定。
