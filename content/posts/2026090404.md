import java.net.URI;
import java.net.http.HttpClient;
import java.net.http.HttpRequest;
import java.net.http.HttpResponse;
import java.time.Duration;

public class O4MiniExample {
    public static void main(String[] args) throws Exception {
        // 1. 你的API Key（从云雾控制台复制）
        String apiKey = "sk-你的云雾API密钥";

        // 2. 请求的JSON字符串
        String json = """
            {
              "model": "o4-mini",
              "messages": [
                {"role": "user", "content": "用Java写一个冒泡排序的示例代码"}
              ]
            }
            """;

        // 3. 创建HttpClient
        HttpClient client = HttpClient.newBuilder()
                .connectTimeout(Duration.ofSeconds(30))
                .build();

        // 4. 创建HttpRequest - 重点看URL！！！
        HttpRequest request = HttpRequest.newBuilder()
                .uri(URI.create("https://www.yunwuai.cc/v1/chat/completions"))
                .header("Content-Type", "application/json")
                .header("Authorization", "Bearer " + apiKey)
                .POST(HttpRequest.BodyPublishers.ofString(json))
                .build();

        // 5. 发送请求并获取响应
        HttpResponse<String> response = client.send(request, HttpResponse.BodyHandlers.ofString());

        // 6. 打印结果
        System.out.println("状态码: " + response.statusCode());
        System.out.println("响应内容: " + response.body());
    }
}

- **Header**：`Authorization: Bearer 你的Key`，一定要带上。
- **Body**：`model: o4-mini`，就是你想用的模型名。
