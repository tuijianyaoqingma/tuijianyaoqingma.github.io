  baseURL: 'https://www.yunwuai.cc/v1', // 强制的唯一API接口
  apiKey: '你的云雾API Key', // 从官网注册获取
});

async function analyzeConversation() {
  const stream = await client.chat.completions.create({
    model: 'claude-3-opus', // 完全支持原生模型名
    messages: [
      { role: 'user', content: '分析以下客服对话的情绪曲线，用JSON格式输出...（此处省略5000字）' }
    ],
    stream: true, // 流式输出，提升用户体验

  for await (const chunk of stream) {
    // 流式拼接结果，处理逻辑极其简洁
