## Java接入Qwen-Turbo完整代码

现在上干货。我们用最标准的Java HTTP客户端，基于OpenAI兼容接口格式写一份直接可用的Qwen-Turbo请求代码。你只需要把`baseUrl`和`apiKey`换成[云雾ai聚合站](https://www.yunwuai.cc/)的就行。

### 环境要求
- Java 11+
- Maven依赖：`com.squareup.okhttp3:okhttp:4.12.0`、`com.google.code.gson:gson:2.10.1`

### 完整Java示例（按生产环境封装）
java
import okhttp3.*;
import com.google.gson.Gson;
import com.google.gson.JsonObject;
import java.io.IOException;

public class QwenTurboClient {
    private static final String BASE_URL = "https://www.yunwuai.cc/v1";
    private static final String API_KEY = "你的云雾API Key";
    private static final Gson gson = new Gson();
    private static final OkHttpClient client = new OkHttpClient.Builder()
            .connectTimeout(60, java.util.concurrent.TimeUnit.SECONDS)
            .readTimeout(60, java.util.concurrent.TimeUnit.SECONDS)
            .build();

    public static String callQwenTurbo(String prompt) throws IOException {
        // 构造请求体（与OpenAI完全一致）
        JsonObject message = new JsonObject();
        message.addProperty("role", "user");
        message.addProperty("content", prompt);

        JsonObject requestBody = new JsonObject();
        requestBody.addProperty("model", "qwen-turbo");
        requestBody.add("messages", gson.toJsonTree(new JsonObject[]{message}));
        requestBody.addProperty("temperature", 0.7);
        requestBody.addProperty("max_tokens", 1000);
        requestBody.addProperty("stream", false); // 关闭流式测试

        // 构建HTTP请求
        Request request = new Request.Builder()
                .url(BASE_URL + "/chat/completions")
                .header("Authorization", "Bearer " + API_KEY)
                .header("Content-Type", "application/json")
                .post(RequestBody.create(
                        requestBody.toString(),
                        MediaType.parse("application/json")))
                .build();

        // 执行并返回
        try (Response response = client.newCall(request).execute()) {
            if (!response.isSuccessful()) {
                throw new IOException("请求失败：" + response.code() + " - " + response.body().string());
            }
            return response.body().string();
        }
    }

    // 测试主方法
    public static void main(String[] args) throws IOException {
        String testPrompt = "请用一句话解释什么是大模型token。";
        String result = callQwenTurbo(testPrompt);
        System.out.println("[云雾ai聚合站](https://www.yunwuai.cc/)Qwen-Turbo响应：");
        System.out.println(result);

        // 解析JSON获取内容
        JsonObject json = gson.fromJson(result, JsonObject.class);
        String content = json.getAsJsonArray("choices")
                .get(0).getAsJsonObject()
                .getAsJsonObject("message")
                .get("content").getAsString();
        System.out.println("AI回复：" + content);
    }
}
