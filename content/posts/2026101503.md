import os
from openai import OpenAI

# --- 安全配置区域（生产环境必读）---
# 强烈建议使用环境变量设置，而非硬编码。
# 使用前请先执行：export YUNWU_API_KEY="你的云雾API密钥"

    api_key = input("请直接输入你的API密钥（仅供测试，请勿泄露）: ")

# --- 核心配置 ---
client = OpenAI(
    # 核心要点：将默认的OpenAI地址替换为云雾API的国内直连地址
    base_url="https://www.yunwuai.cc/v1",
    api_key=api_key,

# --- 一个生产级的企业对话示例 ---
# 这里模拟了一个“智能客服助手”
system_prompt = """你是Qwen3，一个由阿里云开发的高效AI助手。
你正在协助一家跨境电商公司处理客户咨询。请用专业且友好的语气回复。
如果遇到无法确认的订单信息，请引导客户提供订单号。"""

user_question = "我想查一下昨天下的订单，一直没有更新物流信息，怎么办？"

try:
    # 调用Qwen3模型
    response = client.chat.completions.create(
        model="Qwen3-70B-Chat", # 云雾平台支持的Qwen3模型标识
        messages=[
            {"role": "system", "content": system_prompt},
            {"role": "user", "content": user_question},
        ],
        temperature=0.7,  # 企业级应用建议0.5-0.7之间
        max_tokens=1024,
        stream=False, # 这里设置为False展示完整结果，你也可改为True体验流式输出
    )

    # 提取并打印回复
    answer = response.choices[0].message.content
    print("Qwen3回复如下：")
    print(answer)

except Exception as e:
    # 基础的错误处理
    print(f"调用API时发生错误: {e}")
    print("请检查你的网络连接或API Key是否正确配置。")
