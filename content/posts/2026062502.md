import com.theokanning.openai.OpenAiService;
import com.theokanning.openai.completion.chat.*;

import java.time.Duration;
import java.util.List;

public class GPT4oMiniExample {
    public static void main(String[] args) {
        // 1. 替换成你的[云雾api聚合站](https://www.yunwuai.cc/)API Key
        String apiKey = "sk-你的云雾API密钥";

        // 3. 创建OpenAiService对象，设置超时时间（10秒）
        OpenAiService service = new OpenAiService(apiKey, Duration.ofSeconds(10), baseUrl);
        
        // 4. 组装消息内容
        ChatMessage systemMessage = new ChatMessage(ChatMessageRole.SYSTEM, "你是一个高效的Java技术助手。");
        ChatMessage userMessage = new ChatMessage(ChatMessageRole.USER, "用最简单的Java代码，给我写一个快速排序算法，并加上注释。");

                .messages(List.of(systemMessage, userMessage))
                .temperature(0.7)
                .maxTokens(500)
                .build();
        
        // 6. 发送请求并获取回复（同步方式，流式输出代码略复杂，这里用最直观的）
        try {
            ChatCompletionResult result = service.createChatCompletion(request);
            ChatMessage reply = result.getChoices().get(0).getMessage();
            System.out.println("GPT-4o mini回复：");
            System.out.println(reply.getContent());
        } catch (Exception e) {
            System.err.println("调用失败，错误信息：" + e.getMessage());
            // 常见错误排查在后面
        }
    }
}

| 报错现象                             | 原因           | 解决办法                                                                                          |
| ------------------------------------ | ---------------- | --------------------------------------------------------------------------------------------------- |
| `Network is unreachable`            | 网络不通或超时 | 检查是否能直接访问 `https://www.yunwuai.cc`，不能的话，可能是本地DNS或防火墙问题，换一个网络试试。 |

| `404 Not Found`                     | 模型名称错误   | 确认你输入的是`gpt-4o-mini`，不要多写或少写字符。云雾模型列表中可能有不同版本，写准确即可。      |

## 流式调用（Streaming）：实时获取回复

如果你的应用需要更快的用户体验（比如逐字打印回复），可以用流式方式。代码稍微复杂一点，但效果好很多。下面也给你一个示例：

java
// 流式调用
Flowable<ChatCompletionChunk> flowable = service.streamChatCompletion(request);
flowable.subscribe(chunk -> {
    if (chunk.getChoices() != null && !chunk.getChoices().isEmpty()) {
        String delta = chunk.getChoices().get(0).getMessage().getContent();
        if (delta != null) {
            System.out.print(delta); // 逐字输出
        }
    }
});
