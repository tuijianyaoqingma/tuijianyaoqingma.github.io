   创建一个`index.js`文件，内容如下（完整可运行）：

javascript
const OpenAI = require('openai');

const client = new OpenAI({
  apiKey: '你的API_KEY',  // 替换成你从云雾平台获取的key
  baseURL: 'https://www.yunwuai.cc/v1' // 关键：改成云雾的地址
});

async function main() {
  try {
    const response = await client.chat.completions.create({
      model: 'gpt-5-thinking', // 云雾支持的模型名，这里用GPT-5-thinking
      messages: [
        { role: 'user', content: '用中文写一段关于人工智能的简介' }
      ],
      temperature: 0.7,
      max_tokens: 500
    });
    console.log(response.choices[0].message.content);
  } catch (error) {
    console.error('调用失败：', error);
  }
}

main();
