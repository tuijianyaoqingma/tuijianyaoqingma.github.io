### 第二步：一段独立的多模型 Java 调用逻辑

如果你使用 Spring Boot，以下代码核心模块即可实现无缝切换。这里我们以三个模型为例，将 switch 逻辑写进一个服务方法里：

java
@Component
public class AiModelService {

    private static final String BASE_URL = "https://www.yunwuai.cc/v1";
    private static final String API_KEY = "sk-你的云雾API_Key"; 

    @Autowired
    private RestTemplate restTemplate;

    @Value("${ai.model.name:qwen-turbo}") // 默认使用Qwen
    private String currentModel;

    public String chat(String messages) {
        Map<String, Object> body = new HashMap<>();
        body.put("model", currentModel); // 核心：只改这个字段
        body.put("messages", List.of(Map.of("role", "user", "content", messages)));
        
        HttpHeaders headers = new HttpHeaders();
        headers.set("Authorization", "Bearer " + API_KEY);
        headers.setContentType(MediaType.APPLICATION_JSON);
        
        HttpEntity<Map<String, Object>> request = new HttpEntity<>(body, headers);
        
        ResponseEntity<Map> response = restTemplate.postForEntity(BASE_URL + "/chat/completions", request, Map.class);
        // 这里解析 response，拿到 AI 回复
        return extractResponseText(response.getBody());
    }

    // 切换模型的方法
    public void switchModel(String modelName) {
        this.currentModel = modelName;
        System.out.println("当前模型已切换到：" + modelName);
    }
}
